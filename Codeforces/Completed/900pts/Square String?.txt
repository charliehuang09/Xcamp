#include <bits/stdc++.h>
using namespace std;
string solve(string input){
    if (input.size() % 2 == 1){
        return "NO";
    }
    string a = input.substr(0, input.size() / 2);
    string b = input.substr(input.size() / 2, input.size() / 2);
    if (a == b){
        return "YES";
    }
    return "NO";

}
int main(){
    int length;
    string t;
    cin >> length;
    string output[length];
    for (int i = 0; i < length; i++){
        cin >> t;
        solve (t);
        output[i] = solve(t);
    }
    cout << "\n";
    for (int i = 0; i < length; i++){
        cout << output[i] << "\n";
    } 
}

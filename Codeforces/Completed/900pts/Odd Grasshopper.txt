#include <bits/stdc++.h>
using namespace std;
int main(){
    // cout << 5 % 4;
    // return 0;
    int testcases;
    cin >> testcases;
    for (int i = 0; i < testcases; i++){
        cout << "\n";
        long long int pos, times;
        cin >> pos >> times;
        for (long long int j = times - (times % 4) + 1; j <= times; j++){
            if (pos % 2 == 0){
                pos -= j;
            }
            else{
                pos += j;
            }
        }
        cout << pos;
    }
}
